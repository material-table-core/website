Styling Rows#

Props#

FieldTypeDescription
options.rowStylestyle object or funcstyle to apply in all rows cells

Usage#

import MaterialTable from "@material-table/core";
<MaterialTable
// ...
options={{
rowStyle: {
backgroundColor: "#6ABAC9",
},
}}
/>;
